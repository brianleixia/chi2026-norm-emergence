\subsection{Primary Outcome: Behavioral Convergence Score (BCS)}
\label{sec:bcs}

\begin{equation}
\text{BCS}(t) = \max_i p_i(t)
\end{equation}

where $p_i(t)$ is the proportion of agents choosing behavioral category $i$ at round $t$. \textbf{Norm emergence threshold}: $\text{BCS} \geq 0.70$ sustained for $\geq 3$ consecutive rounds.

\subsection{Statistical Analysis}
